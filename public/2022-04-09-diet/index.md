import os
import zipfile

def convert_to_json(year):
  input = str(year)
  with open(input+'.csv','r') as f:
      fl = f.readlines()
      result = []
      for _ in fl:
          a = list(_)
          a = a[5:-1]
          a.insert(5,'"')
          a.insert(0,'"')
          _ = ''.join(a)
          _ = _.strip('\n')
          result.append('['+f'{_}'+'],')
      result[-1] = result[-1].strip(',')
      fw = open(input+".txt","w")
      fw.writelines('[')
      fw.writelines(result)
      fw.writelines(']')
      fw.close()
